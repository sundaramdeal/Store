<script>

const WHATSAPP="916393884280";
const DELIVERY=20;

let products=[
{id:1,name:"Parle G Biscuit",price:5},
{id:2,name:"Kurkure Masala",price:10},
{id:3,name:"Lays Chips",price:20},
{id:4,name:"Dairy Milk",price:40},
{id:5,name:"Mango Bite",price:1}
];

let cart=[];

function renderProducts(){

let search=document.getElementById("search").value.toLowerCase();

let html="";

products
.filter(p=>p.name.toLowerCase().includes(search))
.forEach(p=>{

let item=cart.find(x=>x.id==p.id);
let qty=item?item.qty:0;

html+=`
<div class="card">

<h3>${p.name}</h3>

<div class="price">₹${p.price}</div>

${qty==0?

`<button onclick="increase(${p.id})">Add To Cart</button>`

:

`<div class="qtyBox">
<button onclick="decrease(${p.id})">➖</button>
<span>${qty}</span>
<button onclick="increase(${p.id})">➕</button>
</div>`

}

</div>
`;

});

document.getElementById("productList").innerHTML=html;

}

function increase(id){

let p=products.find(x=>x.id==id);
let item=cart.find(x=>x.id==id);

if(item){

item.qty++;

}else{

cart.push({
id:p.id,
name:p.name,
price:p.price,
qty:1
});

}

updateCart();
renderProducts();

}

function decrease(id){

let item=cart.find(x=>x.id==id);

if(!item) return;

item.qty--;

if(item.qty<=0){

cart=cart.filter(x=>x.id!=id);

}

updateCart();
renderProducts();

}

function updateCart(){

document.getElementById("cartCount").innerText=cart.length;

let html="";
let itemsTotal=0;

cart.forEach((item,i)=>{

itemsTotal+=item.price*item.qty;

html+=`
<div class="cartItem">

<div>${item.name}</div>

<div>

<button onclick="decrease(${item.id})">➖</button>
${item.qty}
<button onclick="increase(${item.id})">➕</button>

₹${item.price*item.qty}

<button onclick="removeItem(${i})">❌</button>

</div>

</div>
`;

});

let finalTotal=itemsTotal+DELIVERY;

document.getElementById("cartItems").innerHTML=html;
document.getElementById("itemsTotal").innerText=itemsTotal;
document.getElementById("deliveryFee").innerText=DELIVERY;
document.getElementById("totalPrice").innerText=finalTotal;

}

function removeItem(i){

cart.splice(i,1);

updateCart();
renderProducts();

}

function openCart(){

document.getElementById("cartPage").style.display="block";

}

function closeCart(){

document.getElementById("cartPage").style.display="none";

}

function placeOrder(){

let name=document.getElementById("name").value;
let phone=document.getElementById("phone").value;
let village=document.getElementById("village").value;
let post=document.getElementById("post").value;
let district=document.getElementById("district").value;
let address=document.getElementById("address").value;

if(!name || !phone || !village || !post || !district || !address){

alert("Please fill all delivery details");

return;

}

let itemsTotal=0;

cart.forEach(item=>{
itemsTotal+=item.price*item.qty;
});

if(itemsTotal<50){

alert("Minimum order ₹50 hona chahiye");

return;

}

let total=itemsTotal+DELIVERY;

let msg="🛒 New Order%0A%0A";

msg+="Name: "+name+"%0A";
msg+="Phone: "+phone+"%0A";
msg+="Village: "+village+"%0A";
msg+="Post: "+post+"%0A";
msg+="District: "+district+"%0A";
msg+="Address: "+address+"%0A%0A";

msg+="Order Items:%0A";

cart.forEach(item=>{
msg+=item.name+" x "+item.qty+" = ₹"+(item.price*item.qty)+"%0A";
});

msg+="%0AItems Total: ₹"+itemsTotal;
msg+="%0ADelivery Fee: ₹"+DELIVERY;
msg+="%0AFinal Total: ₹"+total;

let url="https://wa.me/"+WHATSAPP+"?text="+msg;

window.open(url,"_blank");

cart=[];

updateCart();
renderProducts();

closeCart();

}

document.getElementById("search").addEventListener("input",renderProducts);

renderProducts();

</script>
